40static inline std::string getCurrentTime()
41{
42 // Set up boost time zone adjustment and time facet
43 using local_adj =
44 boost::date_time::c_local_adjustor<boost::posix_time::ptime>;
45 boost::posix_time::time_facet* f = new boost::posix_time::time_facet();
46 f->time_duration_format("%+%H:%M");
47
48 // get local time, utc time, and offset
49 auto now = boost::posix_time::microsec_clock::universal_time();
50 auto utc_now = local_adj::utc_to_local(now);
51 boost::posix_time::time_duration td = utc_now - now;
52
53 // Format the date and time in ISO 8601 format with the UTC offset
54 std::ostringstream oss_offset;
55 oss_offset.imbue(std::locale(oss_offset.getloc(), f));
56 oss_offset << td;
57
58 std::string currentTime = to_iso_extended_string(utc_now);
59 currentTime += oss_offset.str();
60
61 return currentTime;
62}

91static inline std::string mergePaths(const std::string& path1,
92 const std::string& path2)
93{
94 std::string result = path1;
95 // Remove trailing "/" from path1
96 while (!result.empty() && result.back() == '/' && result != "/") {
97 result.pop_back();
98 }
99 // Remove leading "/" from path2
100 size_t start = 0;
101 while (start < path2.size() && path2[start] == '/') {
102 start++;
103 }
104 // Get path2 without trailing slashes
105 std::string path2Clean = path2.substr(start);
106 while (!path2Clean.empty() && path2Clean.back() == '/' && path2Clean != "/") {
107 path2Clean.pop_back();
108 }
109 // Append path2 to path1 with a "/" in between
110 if (!result.empty() && !path2Clean.empty()) {
111 result += '/';
112 }
113 result += path2Clean;
114
115 // Remove any potential occurrences of "//" and replace with "/"
116 size_t pos = result.find("//");
117 while (pos != std::string::npos) {
118 result.replace(pos, 2, "/");
119 pos = result.find("//", pos);
120 }
121
122 // Remove trailing "/" from final result if not root path
123 while (!result.empty() && result.back() == '/' && result != "/") {
124 result.pop_back();
125 }
126
127 return result;
128}

138static inline void convertFloatToInt16LE(const float* source,
139 void* dest,
140 SizeType numSamples)
141{
142 // TODO - several steps in this function may be unnecessary for our use
143 // case. Consider simplifying the intermediate cast to char and the
144 // final cast to uint16_t.
145 auto maxVal = static_cast<double>(0x7fff);
146 auto intData = static_cast<char*>(dest);
147
148 for (SizeType i = 0; i < numSamples; ++i) {
149 auto clampedValue =
150 std::clamp(maxVal * static_cast<double>(source[i]), -maxVal, maxVal);
151 auto intValue =
152 static_cast<uint16_t>(static_cast<int16_t>(std::round(clampedValue)));
153 intValue = boost::endian::native_to_little(intValue);
154 *reinterpret_cast<uint16_t*>(intData) = intValue;
155 intData += 2; // destBytesPerSample is always 2
156 }
157}

165static inline std::unique_ptr<int16_t[]> transformToInt16(
166 SizeType numSamples, float conversion_factor, const float* data)
167{
168 std::unique_ptr<float[]> scaledData = std::make_unique<float[]>(numSamples);
169 std::unique_ptr<int16_t[]> intData = std::make_unique<int16_t[]>(numSamples);
170
171 // copy data and multiply by scaling factor
172 float multFactor = 1.0f / (32767.0f * conversion_factor);
173 std::transform(data,
174 data + numSamples,
175 scaledData.get(),
176 [multFactor](float value) { return value * multFactor; });
177
178 // convert float to int16
179 convertFloatToInt16LE(scaledData.get(), intData.get(), numSamples);
180
181 return intData;
182}

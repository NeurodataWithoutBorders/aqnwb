202 {
203 assert(index < m_shape[0]);
204 std::array<size_type, NDIMS - 1> sub_shape {};
205 std::array<size_type, NDIMS - 1> sub_strides {};
206 for (size_type i = 1; i < NDIMS; ++i) {

208 sub_strides[i - 1] = m_strides[i];

225 }

234 const DTYPE* end() const

236 assert(m_shape[0] == 0 || m_data != nullptr);
237 if (m_shape[0] == 0) {
238 return m_data;

303 template<std::size_t NDIMS>
304 inline ConstMultiArrayView<DTYPE, NDIMS> as_multi_array() const
305 {
306 if (shape.size() != NDIMS) {
307 throw std::invalid_argument(
308 "Shape size does not match the number of dimensions.");
309 }
310
311 // Calculate the total number of elements expected
312 SizeType expected_size = 1;
313 for (SizeType dim : shape) {
314 expected_size *= dim;
315 }
316
317 if (data.size() != expected_size) {
318 throw std::invalid_argument("Data size does not match the shape.");
319 }
320
321 std::array<std::size_t, NDIMS> shape_array {};
322 for (std::size_t i = 0; i < NDIMS; ++i) {
323 shape_array[i] = static_cast<std::size_t>(shape[i]);
324 }
325
326 std::array<std::size_t, NDIMS> strides {};
327 std::size_t stride = 1;
328 for (std::size_t i = NDIMS; i-- > 0;) {
329 strides[i] = stride;
330 stride *= shape_array[i];
331 }
332
333 return ConstMultiArrayView<DTYPE, NDIMS>(data.data(), shape_array, strides);
334 }

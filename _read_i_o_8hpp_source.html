203 {
204 assert(index < m_shape[0]);
205 std::array<size_type, NDIMS - 1> sub_shape {};
206 std::array<size_type, NDIMS - 1> sub_strides {};

209 sub_strides[i - 1] = m_strides[i];

226 }

235 const DTYPE* end() const
236 {
237 assert(m_shape[0] == 0 || m_data != nullptr);
238 if (m_shape[0] == 0) {
239 return m_data;
240 }

303 template<std::size_t NDIMS>
304 inline ConstMultiArrayView<DTYPE, NDIMS> as_multi_array() const
305 {
306 if (shape.size() != NDIMS) {
307 throw std::invalid_argument(
308 "Shape size does not match the number of dimensions.");
309 }
310
311 // Calculate the total number of elements expected
312 SizeType expected_size = std::accumulate(
313 shape.begin(), shape.end(), SizeType {1}, std::multiplies<SizeType> {});
314
315 if (data.size() != expected_size) {
316 throw std::invalid_argument("Data size does not match the shape.");
317 }
318
319 std::array<std::size_t, NDIMS> shape_array {};
320 for (std::size_t i = 0; i < NDIMS; ++i) {
321 shape_array[i] = static_cast<std::size_t>(shape[i]);
322 }
323
324 std::array<std::size_t, NDIMS> strides {};
325 std::size_t stride = 1;
326 for (std::size_t i = NDIMS; i-- > 0;) {
327 strides[i] = stride;
328 stride *= shape_array[i];
329 }
330
331 return ConstMultiArrayView<DTYPE, NDIMS>(data.data(), shape_array, strides);
332 }

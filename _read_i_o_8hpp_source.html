208 template<size_type N = NDIMS, typename std::enable_if_t<(N > 1), int> = 0>
209 ConstMultiArrayView<DTYPE, NDIMS - 1> operator[](size_type index) const
210 {
211 assert(index < m_shape[0]);
212 std::array<size_type, NDIMS - 1> sub_shape {};
213 std::array<size_type, NDIMS - 1> sub_strides {};
214 for (size_type i = 1; i < NDIMS; ++i) {
215 sub_shape[i - 1] = m_shape[i];
216 sub_strides[i - 1] = m_strides[i];
217 }
218 return ConstMultiArrayView<DTYPE, NDIMS - 1>(
219 m_data + index * m_strides[0], sub_shape, sub_strides);
220 }
221
228 template<size_type N = NDIMS, typename std::enable_if_t<N == 1, int> = 0>
229 const DTYPE* begin() const
230 {
231 assert(m_shape[0] == 0 || m_data != nullptr);
232 return m_data;
233 }
234
241 template<size_type N = NDIMS, typename std::enable_if_t<N == 1, int> = 0>
242 const DTYPE* end() const
243 {
244 assert(m_shape[0] == 0 || m_data != nullptr);
245 if (m_shape[0] == 0) {
246 return m_data;
247 }
248 return m_data + (m_shape[0] * m_strides[0]);
249 }
250
257 const std::array<size_type, NDIMS>& shape() const { return m_shape; }

259private:

310 template<std::size_t NDIMS>
311 inline ConstMultiArrayView<DTYPE, NDIMS> as_multi_array() const
312 {
313 if (shape.size() != NDIMS) {
314 throw std::invalid_argument(
315 "Shape size does not match the number of dimensions.");
316 }
317
318 // Calculate the total number of elements expected
319 SizeType expected_size = std::accumulate(
320 shape.begin(), shape.end(), SizeType {1}, std::multiplies<SizeType> {});
321
322 if (data.size() != expected_size) {
323 throw std::invalid_argument("Data size does not match the shape.");
324 }
325
326 std::array<std::size_t, NDIMS> shape_array {};
327 for (std::size_t i = 0; i < NDIMS; ++i) {
328 shape_array[i] = static_cast<std::size_t>(shape[i]);
329 }
330
331 std::array<std::size_t, NDIMS> strides {};
332 std::size_t stride = 1;
333 for (std::size_t i = NDIMS; i-- > 0;) {
334 strides[i] = stride;
335 stride *= shape_array[i];
336 }
337
338 return ConstMultiArrayView<DTYPE, NDIMS>(data.data(), shape_array, strides);
339 }
